import time
from machine import Pin

led = Pin(5,Pin.OUT)        # create LED object from pin-05 and set pin-05 to output
 
while True:
  led.value(1)              # turn on
  time.sleep(0.5)	    # delay 0.5 secs
  led.value(0)              # turn off
  time.sleep(0.5)	    # delay 0.5 secs
